Pornește Rapid

Creează primul tău circuit cuantic


# Creează un circuit Bell State
from qiskit import QuantumCircuit

# Inițializează un circuit cu 2 qubiți
qc = QuantumCircuit(2, 2)

# Aplică poarta Hadamard pe primul qubit
qc.h(0)

# Aplică CNOT cu qubit 0 ca control și qubit 1 ca țintă
qc.cx(0, 1)

# Măsoară ambii qubiți
qc.measure([0, 1], [0, 1])

# Execută și vizualizează
qc.draw(output='mpl')
